    //结构体指针

#include 
#include 
using  namespace std;

//自定义数据类型，一些类型集合组成的一个类型
struct student{
    string name;
    int age=0;
    int score=0;
};

int main() {
    //创建结构体变量
    struct student s = {"张三",19,99};

    //通过指针指向结构体变量(结构体类型的指针)
    //c++中，创建结构体变量的时候可以省略struct关键字
    student *p = &s;

    //利用指针访问结构体变量的内容
    //通过结构体指针访问结构体内的属性，需要用"->"运算符来访问
    cout << "姓名：" << p->name << " 年龄："<< p->age << " 成绩:"<< p->score <
    通过结构体指针访问结构体内的属性，需要用"->"运算符来访问
